DROP TABLE IF EXISTS common_code_detail;
DROP TABLE IF EXISTS common_code_group;

create table common_code_group(
    group_code varchar(50) primary key, 
    group_name varchar(100) not null, 
    
    description varchar(500),
    use_yn char(1) not null default 'Y',
    created_at datetime not null default current_timestamp,
    updated_at datetime not null default current_timestamp on update current_timestamp
);

create table common_code_detail(
    group_code varchar(50),
    code varchar(50),
    name varchar(100),
    primary key (group_code, code),
    foreign key (group_code) references common_code_group(group_code),
    
    sort_order int not null default 0,
    use_yn char(1) not null default 'Y',
    created_at datetime not null default current_timestamp,

* common_code_group
- group_code: identifier
- group_name: display name
- description: it is edited by admin in the admin page. developer and admin ca show, not customer
- use_yn: activated or not

* common_code_detail
- group_code: is included in which group
- code: identifier(ORDER_STATUS, MEMBER_GRADE)
- name: display name
- description: it must be a very detail description. ex. 'in delivery' is what  
- sort_order: display order
- use_yn

pk is composite key(group_code, code). using composite key in common code is a standardized pattern

* use_yn type
- CHAR(1) is used commonly in the past. any db does not support boolean.
- in a recent project, use BOOLEAN instead of CHAR(1)

- query tables with common code
```sql
-- ### 공통 코드와 조인하여 조회
SELECT
  m.member_id,
  m.name,
  m.email,
  m.grade,
  c.name AS grade_name
FROM members m
JOIN common_code_detail c
  ON c.group_code = 'MEMBER_GRADE' AND m.grade = c.code
ORDER BY m.member_id;

SELECT
  o.order_id,
  m.name AS member_name,
  o.order_status,
  c.name AS status_name,
  o.total_amount
FROM orders o
JOIN members m ON o.member_id = m.member_id
JOIN common_code_detail c
  ON c.group_code = 'ORDER_STATUS' AND o.order_status = c.code
ORDER BY o.order_id;

SELECT
  p.payment_id,
  o.order_id,
  p.payment_method,
  pm.name AS method_name,
  p.payment_status,
  ps.name AS status_name,
  p.amount
FROM payments p
JOIN orders o ON p.order_id = o.order_id
JOIN common_code_detail pm 
    ON pm.group_code = 'PAYMENT_METHOD' AND p.payment_method = pm.code
JOIN common_code_detail ps
    ON ps.group_code = 'PAYMENT_STATUS' AND p.payment_status = ps.code
ORDER BY p.payment_id;
```

- query code list by specific group
```sql
-- ### 특정 그룹의 코드 목록 조회
-- 주문 상태 목록 조회
SELECT code, name
FROM common_code_detail
WHERE group_code = 'ORDER_STATUS'
 AND use_yn = 'Y'
ORDER BY sort_order;

-- 결제 수단 목록 조회
SELECT code, name
FROM common_code_detail
WHERE group_code = 'PAYMENT_METHOD'
 AND use_yn = 'Y'
ORDER BY sort_order;
```
