* many-to-many relationship - junction table

- in rdbms, resolve m:n relationship by using junction table
- m:n -> 1:n, n:1
```sql
create table orders(
    order_id int primary key
);
create table product(
    product_id int primary key
);
create table order_product(
    order_product_id int,
    order_id int,
    product_id int,
    constraint unique_order_product unique(order_id, product_id),
    constraint fk_order_product_orders foreign key(order_id) references orders(order_id),
    constraint fk_order_product_product foreign key(product_id) references product(product_id)
);
```
