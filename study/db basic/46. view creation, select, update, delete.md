* view creation, select, update, delete
- CREATE VIEW: view creation
`CREATE VIEW view_name AS SELECT query`
- practice: use prefix view_, v_
```sql
drop view if exists v_category_order_status;

create view v_category_order_status as -- create view
select o.order_id, o.status, o.user_id
from orders o;

alter view v_category_order_status as  -- alter view
select o.order_id, o.status
from orders o;

drop view v_category_order_status; -- drop view

select * from v_category_order_status; -- query view
```
